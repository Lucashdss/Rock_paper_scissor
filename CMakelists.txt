cmake_minimum_required(VERSION 3.14)

project(generalTestProject)

# Dependencies
find_package(Catch2 3 REQUIRED)
find_package(SFML CONFIG REQUIRED COMPONENTS Graphics Audio Window System Network)

# Executable
add_executable(generalTest
    src/player/player.cpp
    src/player/playerTest.cpp
    src/player/cpuEnemy/cpuEnemy.cpp
    src/player/cpuEnemy/cpuEnemyTest.cpp
    src/animation/animation.cpp
    src/animation/animationTest.cpp
)

# Include directories
target_include_directories(generalTest PRIVATE
    src/player
    src/player/cpuEnemy
    src/animation
)

# Link dependencies
target_link_libraries(generalTest PRIVATE
    Catch2::Catch2WithMain
    SFML::Graphics
    SFML::Audio
    SFML::Window
    SFML::System
    SFML::Network
)
