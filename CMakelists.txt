cmake_minimum_required(VERSION 3.14)  # use a modern minimum version

project(PlayerTestProject)

# Find Catch2 (you must have Catch2 v3 installed and discoverable)
find_package(Catch2 3 REQUIRED)

# Create your test executable — include all relevant source files
add_executable(playerTest
    src/player/player.cpp
    src/player/playerTest.cpp
    src/player/cpuEnemy/cpuEnemy.cpp
    src/player/cpuEnemy/cpuEnemyTest.cpp
)

# Link against Catch2’s testing main and headers
target_link_libraries(playerTest PRIVATE Catch2::Catch2WithMain)

# Add include directories so headers like "player.hpp" resolve
target_include_directories(playerTest PRIVATE
    src/player
    src/player/cpuEnemy
)
